# Spec W6.4 — Paper-trader observability port

**Назначение spec:** для агента-исполнителя (cheap LLM). Выполнить последовательно, не отступая ни от одного шага.

**Контекст продукта:** `solana-alpha`. Репо: `c:/Users/cente/Ideas/solana-alpha`. Ветка `v2`. HEAD после W6.3c: `7b9bc6b1be0e6dc75605ac74092253529f14e35a`.

**Время выполнения:** ~3 часа. Это «portoring» этап: код в `_salvage_legacy/scripts-tmp/` уже написан и проверен, нужно скопировать с минимальными правками под v2 (env-имена, `SA_PG_DSN`, `scripts/lib/telegram.mjs`, схема W6.3c JSONL).

## Цель

После W6.3c paper-trader открывает позиции и пишет полные `close` события с cost breakdown. **Но мы наблюдаем за ним только через `pm2 logs` и `cat *.jsonl`.** W6.4 даёт человеку (и cron'ам) UI и алерты:

1. **Dashboard `/papertrader2`** — HTML страница (Tailwind+Alpine) поверх существующего `dashboard-organizer-paper` PM2-app, читает все `*.jsonl` из `PAPER2_DIR=/opt/solana-alpha/data/paper2/`.
2. **Endpoint `/api/paper2`** — JSON-агрегат всех стратегий (open/closed counts, pnl USD, win rate, exits breakdown, recent closes). Использует **`close.netPnlUsd`** напрямую (W6.3c контракт), а не псевдо-`pctToUsd(pnlPct)` как в legacy.
3. **Hourly Telegram report** (`scripts-tmp/hourly-telegram-report.mjs`) — каждый час шлёт сводку в TG: по стратегиям, лучшие/худшие закрытия, coverage по DEX, health таблиц.
4. **Healthcheck** (`scripts-tmp/paper2-healthcheck.mjs`) — каждые 10 мин: если хоть один `*.jsonl` не обновлялся > `PAPER2_MAX_STALE_MIN` минут — TG-алерт.
5. **Advisor** (`scripts-tmp/paper2-advisor.mjs`) — каждые 3 часа: ранжирует стратегии по `score=sum_pnl + 0.5*max_drawdown`, советует mint add/drop, exit-tweak suggestions. Пишет `data/paper2/advisor-latest.json` + `data/advisor/journal.jsonl`. Шлёт в TG.
6. **Advisor digest** (`scripts-tmp/advisor-digest.mjs`) — раз в сутки: суточная сводка по advisor journal'у, persistent ideas vs one-off.
7. **Post-mortem** (`scripts-tmp/post-mortem-paper-v1.mjs`) — on-demand утилита: разбивает закрытия по age/peak/exit-reason, проверяет миграцию мина в Raydium/Meteora/Orca через JOIN. Не cron, запускается вручную из shell.
8. **Logrotate** для JSONL: каждые сутки rotate + gzip, чтобы JSONL'ы не росли неограниченно.

После W6.4: 11 PM2 процессов (без изменений), новые HTTP эндпоинты в `dashboard-organizer-paper` (port 3008), 3 cron записи под `salpha`, 1 logrotate конфиг в `/etc/logrotate.d/`.

### 2.2. Хелперы — поместить **перед** `app.listen(...)`

```typescript
// ---------------------------------------------------------
// /api/paper2 — read every *.jsonl in PAPER2_DIR and aggregate.
// Uses W6.3c close.netPnlUsd directly (NOT pctToUsd(pnlPct)),
// because executor stores per-DEX cost breakdown.
// ---------------------------------------------------------
type Paper2OpenItem = {
  mint: string;
  symbol: string;
  entryTs: number;
  entryMcUsd: number;
  entryRealMcUsd: number | null;
  openedAtIso: string | null;
  lane: string | null;
  source: string | null;
  metricType: string | null;
  features: unknown;
  btc: unknown;
  peakMcUsd: number;
  peakPnlPct: number;
  trailingArmed: boolean;
  totalInvestedUsd: number;
  pnlPct?: number | null;
  pnlUsd?: number | null;
  ageMin?: number;
  hasLiveMc?: boolean;
  currentMcUsd?: number;
};

function loadPaper2File(filePath: string): {
  open: Paper2OpenItem[];
  closed: any[];
  firstTs: number;
  lastTs: number;
  resetTs: number;
  evals1h: number;
  passed1h: number;
  failReasons: Array<{ reason: string; count: number }>;
} {
  if (!fs.existsSync(filePath)) {
    return { open: [], closed: [], firstTs: Date.now(), lastTs: Date.now(), resetTs: 0, evals1h: 0, passed1h: 0, failReasons: [] };
  }
  const lines = fs.readFileSync(filePath, 'utf-8').split('\n').filter(Boolean);
  const om = new Map<string, Paper2OpenItem>();
  const cl: any[] = [];
  let f = Date.now();
  let l = 0;
  let resetTs = 0;
  let evals1h = 0;
  let passed1h = 0;
  const failReasonsCount = new Map<string, number>();
  const since1h = Date.now() - 3_600_000;
  for (const ln of lines) {
    let e: any;
    try { e = JSON.parse(ln); } catch { continue; }
    if (typeof e.ts === 'number') {
      if (e.ts < f) f = e.ts;
      if (e.ts > l) l = e.ts;
    }
    if (e.kind === 'reset') { resetTs = e.ts || 0; continue; }
    if (e.kind === 'eval' && (e.ts || 0) >= since1h) {
      evals1h++;
      if (e.pass) passed1h++;
      else {
        for (const r of (e.reasons || [])) {
          failReasonsCount.set(r, (failReasonsCount.get(r) || 0) + 1);
        }
      }
    }
    if (e.kind === 'open') {
      const featMc = (e.features && (e.features.market_cap_usd || e.features.fdv_usd)) || 0;
      om.set(e.mint, {
        mint: e.mint,
        symbol: e.symbol,
        entryTs: e.entryTs,
        entryMcUsd: Number(e.entryMcUsd || 0),
        entryRealMcUsd: featMc || null,
        openedAtIso: e.entryTs ? new Date(e.entryTs).toISOString() : null,
        lane: e.lane || null,
        source: e.source || null,
        metricType: e.metricType || null,
        features: e.features || null,
        btc: e.btc || null,
        peakMcUsd: Number(e.entryMcUsd || 0),
        peakPnlPct: 0,
        trailingArmed: false,
        totalInvestedUsd: Number(e.totalInvestedUsd ?? e.entryMcUsd ?? 0),
      });
    } else if (e.kind === 'peak' && om.has(e.mint)) {
      const o = om.get(e.mint)!;
      o.peakMcUsd = Math.max(o.peakMcUsd, Number(e.peakMcUsd || 0));
      o.peakPnlPct = Math.max(o.peakPnlPct, Number(e.peakPnlPct || 0));
      o.trailingArmed = o.trailingArmed || !!e.trailingArmed;
    } else if (e.kind === 'close') {
      om.delete(e.mint);
      cl.push(e);
    }
  }
  const failReasons = [...failReasonsCount.entries()]
    .sort((a, b) => b[1] - a[1])
    .slice(0, 8)
    .map(([reason, count]) => ({ reason, count }));
  return { open: [...om.values()], closed: cl, firstTs: f, lastTs: l, resetTs, evals1h, passed1h, failReasons };
}

function paper2Metrics(closed: any[]): {
  total: number;
  wins: number;
  winRate: number;
  sumPnlUsd: number;
  avgPnl: number;
  avgPeak: number;
  bestPnlUsd: number;
  worstPnlUsd: number;
  exits: Record<string, number>;
  exitsBreakdown: Record<string, { count: number; sumPct: number; sumUsd: number; avgPct: number }>;
} {
  const exitKinds = ['TP', 'SL', 'TRAIL', 'TIMEOUT', 'NO_DATA', 'KILLSTOP'] as const;
  const exits: Record<string, number> = Object.fromEntries(exitKinds.map((k) => [k, 0]));
  const breakdown: Record<string, { count: number; sumPct: number; sumUsd: number; avgPct: number }> = Object.fromEntries(
    exitKinds.map((k) => [k, { count: 0, sumPct: 0, sumUsd: 0, avgPct: 0 }]),
  );
  if (!closed.length) {
    return { total: 0, wins: 0, winRate: 0, sumPnlUsd: 0, avgPnl: 0, avgPeak: 0, bestPnlUsd: 0, worstPnlUsd: 0, exits, exitsBreakdown: breakdown };
  }
  let sumPct = 0;
  let sumPeak = 0;
  let wins = 0;
  let bestUsd = -Infinity;
  let worstUsd = Infinity;
  let sumUsd = 0;
  for (const c of closed) {
    const pnlPct = Number(c.pnlPct || 0);
    const pnlUsd = Number(c.netPnlUsd ?? (POSITION_USD_DEFAULT * pnlPct) / 100);
    sumPct += pnlPct;
    sumUsd += pnlUsd;
    sumPeak += Number(c.peakPnlPct || 0);
    if (pnlPct > 0) wins++;
    if (pnlUsd > bestUsd) bestUsd = pnlUsd;
    if (pnlUsd < worstUsd) worstUsd = pnlUsd;
    const r: string = c.exitReason || 'NO_DATA';
    if (exits[r] != null) exits[r]++;
    if (breakdown[r]) {
      breakdown[r].count++;
      breakdown[r].sumPct += pnlPct;
      breakdown[r].sumUsd += pnlUsd;
    }
  }
  for (const k of Object.keys(breakdown)) {
    breakdown[k].avgPct = breakdown[k].count ? breakdown[k].sumPct / breakdown[k].count : 0;
  }
  return {
    total: closed.length,
    wins,
    winRate: (wins / closed.length) * 100,
    sumPnlUsd: sumUsd,
    avgPnl: sumPct / closed.length,
    avgPeak: sumPeak / closed.length,
    bestPnlUsd: bestUsd === -Infinity ? 0 : bestUsd,
    worstPnlUsd: worstUsd === Infinity ? 0 : worstUsd,
    exits,
    exitsBreakdown: breakdown,
  };
}
```

### 2.3. Routes — добавить **перед** `app.listen(...)`

```typescript
app.get('/papertrader2', async (_req, reply) => {
  reply.header('content-type', 'text/html; charset=utf-8');
  return fs.readFileSync(HTML2_PATH, 'utf-8');
});

app.get('/api/paper2', async (_req, reply) => {
  reply.header('cache-control', 'no-store');
  let files: string[] = [];
  try {
    if (fs.existsSync(PAPER2_DIR)) {
      files = fs
        .readdirSync(PAPER2_DIR)
        .filter((f) => f.endsWith('.jsonl'))
        .map((f) => path.join(PAPER2_DIR, f));
    }
  } catch {
    /* ignore */
  }

  const strategies: any[] = [];
  for (const fp of files) {
    const sid = path.basename(fp, '.jsonl');
    const { open, closed, firstTs, lastTs, resetTs, evals1h, passed1h, failReasons } = loadPaper2File(fp);
    const m = paper2Metrics(closed);
    const startedAt = resetTs || firstTs;
    const closedWithUsd = closed
      .map((c) => ({
        mint: c.mint,
        symbol: c.symbol,
        exitTs: c.exitTs,
        exitReason: c.exitReason,
        pnlPct: Number(c.pnlPct || 0),
        pnlUsd: Number(c.netPnlUsd ?? (POSITION_USD_DEFAULT * Number(c.pnlPct || 0)) / 100),
        durationMin: Number(c.durationMin || 0),
        dex: c.costs && c.costs.dex,
      }))
      .sort((a, b) => (b.exitTs || 0) - (a.exitTs || 0))
      .slice(0, 20);

    const enrichedOpen = open.slice(0, 30).map((ot) => ({
      mint: ot.mint,
      symbol: ot.symbol,
      entryTs: ot.entryTs,
      entryMcUsd: ot.entryMcUsd,
      entryRealMcUsd: ot.entryRealMcUsd,
      openedAtIso: ot.openedAtIso,
      lane: ot.lane,
      source: ot.source,
      metricType: ot.metricType,
      currentMcUsd: ot.peakMcUsd || ot.entryMcUsd || 0,
      peakMcUsd: ot.peakMcUsd,
      peakPnlPct: ot.peakPnlPct,
      trailingArmed: ot.trailingArmed,
      pnlPct: null,
      pnlUsd: null,
      ageMin: (Date.now() - (ot.entryTs || Date.now())) / 60_000,
      hasLiveMc: false,
    }));

    strategies.push({
      strategyId: sid,
      file: fp,
      openCount: open.length,
      closedCount: closed.length,
      startedAt,
      lastTs,
      hoursOfData: (Date.now() - startedAt) / 3_600_000,
      sumPnlUsd: m.sumPnlUsd,
      winRate: m.winRate,
      avgPnl: m.avgPnl,
      avgPeak: m.avgPeak,
      bestPnlUsd: m.bestPnlUsd,
      worstPnlUsd: m.worstPnlUsd,
      unrealizedUsd: 0,
      exits: m.exits,
      exitsBreakdown: m.exitsBreakdown,
      evals1h,
      passed1h,
      failReasons,
      open: enrichedOpen,
      recentClosed: closedWithUsd,
    });
  }
  strategies.sort((a, b) => b.sumPnlUsd - a.sumPnlUsd);

  const totals = strategies.reduce(
    (acc, s) => {
      acc.strategies += 1;
      acc.open += s.openCount;
      acc.closed += s.closedCount;
      acc.sumPnlUsd += s.sumPnlUsd;
      return acc;
    },
    { strategies: 0, open: 0, closed: 0, sumPnlUsd: 0 },
  );

  return { now: Date.now(), paper2Dir: PAPER2_DIR, totals, strategies };
});
```
